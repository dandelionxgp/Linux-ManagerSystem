#### 1.6.3 Axios 封装 `src/api/request.js`

```javascript
import axios from 'axios'
import { ElMessage } from 'element-plus'
import router from '@/router'

const request = axios.create({
  baseURL: '/api',
  timeout: 15000
})

// 请求拦截器：自动附加 Token
request.interceptors.request.use(config => {
  const token = localStorage.getItem('token')
  if (token) {
    config.headers.Authorization = `Bearer ${token}`
  }
  return config
})

// 响应拦截器：统一错误处理
request.interceptors.response.use(
  response => {
    const res = response.data
    if (res.code !== 0) {
      ElMessage.error(res.message || '请求失败')
      return Promise.reject(new Error(res.message))
    }
    return res
  },
  error => {
    if (error.response?.status === 401) {
      localStorage.removeItem('token')
      router.push('/login')
      ElMessage.error('登录已过期，请重新登录')
    } else {
      ElMessage.error(error.message || '网络错误')
    }
    return Promise.reject(error)
  }
)

export default request
```

```javascript
import { defineStore } from 'pinia'
import { ref } from 'vue'
import { loginApi, logoutApi, getCurrentUserApi } from '@/api/auth'

export const useAuthStore = defineStore('auth', () => {
  const user = ref(null)
  const token = ref(localStorage.getItem('token') || '')

  async function login(username, password) {
    const res = await loginApi({ username, password })
    token.value = res.data.token
    localStorage.setItem('token', token.value)
    await fetchUser()
  }

  async function logout() {
    await logoutApi()
    token.value = ''
    user.value = null
    localStorage.removeItem('token')
  }

  async function fetchUser() {
    const res = await getCurrentUserApi()
    user.value = res.data
  }

  return { user, token, login, logout, fetchUser }
})
```

#### AuthController 核心代码

```php
<?php

namespace App\Http\Controllers\Api;

use App\Http\Controllers\Controller;
use App\Models\User;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\Hash;
use Illuminate\Validation\ValidationException;

class AuthController extends Controller
{
    // 登录
    public function login(Request $request)
    {
        $request->validate([
            'username' => 'required|string',
            'password' => 'required|string',
        ]);

        $user = User::where('username', $request->username)->first();

        if (!$user || !Hash::check($request->password, $user->password)) {
            throw ValidationException::withMessages([
                'username' => ['用户名或密码错误'],
            ]);
        }

        if (!$user->status) {
            return response()->json([
                'code' => 1003,
                'message' => '账号已被禁用'
            ], 403);
        }

        $token = $user->createToken('api-token')->plainTextToken;
        $user->update(['last_login_at' => now()]);

        return response()->json([
            'code' => 0,
            'message' => '登录成功',
            'data' => [
                'token' => $token,
                'user' => $user->only(['id', 'username', 'real_name', 'role'])
            ]
        ]);
    }

    // 登出
    public function logout(Request $request)
    {
        $request->user()->currentAccessToken()->delete();

        return response()->json([
            'code' => 0,
            'message' => '已退出登录'
        ]);
    }

    // 当前用户信息
    public function me(Request $request)
    {
        return response()->json([
            'code' => 0,
            'data' => $request->user()->only(['id', 'username', 'real_name', 'role'])
        ]);
    }
}
```

### 3.3 入库管理 ⭐ 核心

```php
<?php

namespace App\Services;

use App\Models\Product;
use App\Models\StockIn;
use App\Models\StockInItem;
use App\Models\OperationLog;
use Illuminate\Support\Facades\DB;

class StockService
{
    /**
     * 入库操作
     * 使用数据库事务保证原子性
     */
    public function stockIn(array $data, $userId)
    {
        return DB::transaction(function () use ($data, $userId) {

            // 1. 生成入库单号
            $orderNo = 'RK-' . date('Ymd') . '-' . str_pad(
                StockIn::whereDate('created_at', today())->count() + 1,
                4, '0', STR_PAD_LEFT
            );

            // 2. 创建入库单主记录
            $stockIn = StockIn::create([
                'order_no'     => $orderNo,
                'supplier'     => $data['supplier'],
                'total_amount' => 0,  // 先设0，后面累加
                'operator'     => $data['operator'],
                'remark'       => $data['remark'] ?? '',
            ]);

            $totalAmount = 0;

            // 3. 逐条创建入库明细 + 更新库存
            foreach ($data['items'] as $item) {
                $subtotal = $item['quantity'] * $item['unit_price'];

                StockInItem::create([
                    'stock_in_id' => $stockIn->id,
                    'product_id'  => $item['product_id'],
                    'quantity'    => $item['quantity'],
                    'unit_price'  => $item['unit_price'],
                    'subtotal'    => $subtotal,
                ]);

                // 更新商品库存（增加）
                Product::where('id', $item['product_id'])
                    ->increment('current_stock', $item['quantity']);

                $totalAmount += $subtotal;
            }

            // 4. 更新入库单总金额
            $stockIn->update(['total_amount' => $totalAmount]);

            // 5. 记录操作日志
            OperationLog::create([
                'user_id'     => $userId,
                'action'      => 'create',
                'module'      => 'stock_in',
                'target_id'   => $stockIn->id,
                'description' => "创建入库单 {$orderNo}，共 " . count($data['items']) . " 项商品",
                'ip_address'  => request()->ip(),
            ]);

            return $stockIn->load('items.product');
        });
    }
}
```

```php
/**
 * 出库操作
 * 比入库多一步：库存校验 + 行锁防并发
 */
public function stockOut(array $data, $userId)
{
    return DB::transaction(function () use ($data, $userId) {

        // 1. 逐条校验库存（使用行锁防止并发超卖）
        foreach ($data['items'] as $item) {
            $product = Product::where('id', $item['product_id'])
                ->lockForUpdate()
                ->first();

            if (!$product || $product->current_stock < $item['quantity']) {
                throw new \App\Exceptions\InsufficientStockException(
                    "商品 [{$product->name}] 库存不足！当前库存: {$product->current_stock}，出库数量: {$item['quantity']}"
                );
            }
        }

        // 2. 生成出库单号
        $orderNo = 'CK-' . date('Ymd') . '-' . str_pad(
            StockOut::whereDate('created_at', today())->count() + 1,
            4, '0', STR_PAD_LEFT
        );

        // 3. 创建出库单
        $stockOut = StockOut::create([...]);

        // 4. 逐条创建明细 + 扣减库存
        foreach ($data['items'] as $item) {
            StockOutItem::create([...]);
            Product::where('id', $item['product_id'])
                ->decrement('current_stock', $item['quantity']);
        }

        // 5. 记录操作日志
        OperationLog::create([...]);

        return $stockOut;
    });
}
```

```php
// InventoryService.php
public function confirm(Inventory $inventory, $userId)
{
    return DB::transaction(function () use ($inventory, $userId) {

        foreach ($inventory->items as $item) {
            $diff = $item->actual_qty - $item->system_qty;

            if ($diff > 0) {
                // 盘盈 → 自动生成入库单
                $this->createStockIn($item->product_id, $diff, $userId, $inventory->order_no);
            } elseif ($diff < 0) {
                // 盘亏 → 自动生成出库单
                $this->createStockOut($item->product_id, abs($diff), $userId, $inventory->order_no);
            }

            $item->update(['diff_qty' => $diff]);
        }

        // 更新盘点单状态
        $inventory->update([
            'status' => 3,          // 已确认
            'confirmed_at' => now()
        ]);
    });
}
```

Nginx 配置：

```nginx
server {
    listen 80;
    server_name _;                            # 替换为实际域名
    root /var/www/stock-manager/frontend/dist;
    index index.html;

    # API 反向代理到 Laravel
    location /api {
        proxy_pass http://127.0.0.1:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }

    # Vue SPA 路由（所有非文件请求回退到 index.html）
    location / {
        try_files $uri $uri/ /index.html;
    }

    # 静态资源缓存
    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2)$ {
        expires 30d;
        add_header Cache-Control "public, immutable";
    }
}
```

## 遇到问题怎么办？🔧

| 问题 | 排查方法 |
|------|----------|
| Laravel 500 错误 | `tail -100 backend/storage/logs/laravel.log` |
| 数据库连接失败 | `sudo systemctl status mysql` → 检查 `.env` 配置 |
| API 返回 404 | 检查 `routes/api.php` 是否正确注册 |
| 前端 API 请求 404 | 检查 `vite.config.js` 的 proxy 配置 |
| CORS 跨域错误 | 检查 Laravel CORS 中间件配置 |
| "Target class does not exist" | `composer dump-autoload` |
| Migration 报错 | `php artisan migrate:rollback` → 修正 → `php artisan migrate` |
| Composer 安装慢 | `composer config -g repo.packagist composer https://mirrors.aliyun.com/composer/` |
| npm 安装慢 | `npm config set registry https://registry.npmmirror.com` |
| 端口被占用 | `lsof -i :8000` 或 `lsof -i :5173` → `kill -9 PID` |

## 核心建议 💡

### 给新手的6条黄金法则

1. **先跑通，再优化**：第一天务必让 `php artisan serve` 和 `npm run dev` 同时成功运行

2. **严格按模块顺序开发**：商品 → 入库 → 出库 → 查询 → 盘点。每个模块完整做完再做下一个

3. **每一步都验证**：
   - 后端：用 Postman / Bruno 测试每个 API
   - 前端：在浏览器中操作，观察 Network 标签
   - 数据：用 `mysql -u stockadmin -p123456 stock_manager` 直接查表确认

4. **善用 Laravel 的 DB::transaction()**：所有涉及库存变动的操作必须用事务包裹

5. **遇到报错三步法**：
   - ① 先看错误信息（`laravel.log` 或浏览器 Console）
   - ② 复制关键错误信息去搜索（Google / StackOverflow）
   - ③ 还是解决不了，检查每一步是否遗漏了关键配置

6. **Git 勤提交，小步快跑**：
   ```bash
   git init
   git add .
   git commit -m "feat: 完成商品管理CRUD"
   ```
   每完成一个功能就 commit，出问题可以 `git diff` 看改了什么，或 `git reset` 回退
